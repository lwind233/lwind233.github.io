有一说一，没啥好讲的，就是模拟一下

需要注意的点就是先移动，后判断下一步移动方向

初始位置可以放在矩阵外，方便统一移动方法

```java
class Solution {
    public int[][] generateMatrix(int n) {
        int[][] result = new int[n][n];


        int dire = 0;
        int i=0;
        int j=-1;
        int cnt=0;
        int total = n*n;
        while(cnt<total){
            cnt++;
            if(dire==0){
                j++;
                result[i][j]=cnt;
                if(j==n-1 || result[i][j+1]!=0) dire=1;
            }else if(dire==1){
                i++;
                result[i][j]=cnt;
                if(i==n-1 || result[i+1][j]!=0) dire=2;
            }else if(dire==2){
                j--;
                result[i][j]=cnt;
                if(j==0 || result[i][j-1]!=0) dire=3;
            }else if(dire==3){
                i--;
                result[i][j]=cnt;
                if(i==0 || result[i-1][j]!=0) dire=0;
            }
        }
        
        return result;


    }
}
```
