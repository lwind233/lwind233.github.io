每一步的开头是将当前元素入结果列表，然后尝试向前走一步判断是否在矩阵里或者是否重复，若不符合条件就回退一步，转个方向重新走，确保每一步的开头都是合法位置

```java
class Solution {
    public List<Integer> spiralOrder(int[][] matrix) {
        List<Integer> result = new ArrayList<>();
        int dire=0;
        int h = matrix.length;
        int w = matrix[0].length;
        int[][] used = new int[h][w];
        int i=0;
        int j=0;
        int[][] dires = new int[4][2];
        dires[0][0] = 0;
        dires[0][1] = 1;

        dires[1][0] = 1;
        dires[1][1] = 0;
        
        dires[2][0] = 0;
        dires[2][1] = -1;
        
        dires[3][0] = -1;
        dires[3][1] = 0;

        while(true){
            result.add(matrix[i][j]);
            if(result.size()==h*w) break;

            used[i][j]=1;
            i += dires[dire][0];
            j += dires[dire][1];
            if(i<0 || i>=h || j<0 || j>=w || used[i][j]==1){
                i -= dires[dire][0];
                j -= dires[dire][1];
                dire = (dire+1)%4;
                i += dires[dire][0];
                j += dires[dire][1];
            }
        }
        return result;

    }
}
```
