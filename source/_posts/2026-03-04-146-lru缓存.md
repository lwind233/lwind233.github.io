这道题不细讲了，流程很简单就是维护一个双向链表和一个哈希表，要点在于要用哑头和哑尾来避免null的问题，否则判断会非常麻烦

```java
class LRUCache {

    class DDQ{
        int val;
        int key;
        DDQ next;
        DDQ prev;

        DDQ(int key, int val){
            this.val = val;
            this.key = key;
        }

        DDQ(int key, int val, DDQ next, DDQ prev){
            this.key = key;
            this.val  = val;
            this.next = next;
            this.prev = prev;
        }
    }

    Map<Integer, DDQ> lMap = new LinkedHashMap<>();

    int length = 0;
    int capacity = 0;
    DDQ dHead = new DDQ(-1,-1);
    DDQ dTail = new DDQ(-1,-1);



    public LRUCache(int capacity) {
        this.capacity = capacity;
        this.dHead.next = this.dTail;
        this.dTail.prev = this.dHead;
    }
    
    public int get(int key) {
        if(lMap.containsKey(key)){
            int value = lMap.get(key).val;
            put(key,value);
            return value;
        } 
        else return -1;
    }
    
    public void put(int key, int value) {
        if(!lMap.containsKey(key)){
            if(this.length==capacity){
                int rKey = dHead.next.key;
                lMap.remove(rKey);
                dHead.next = dHead.next.next;
                if(dHead.next!=null) dHead.next.prev = dHead;
                length--;
            }
        }else{
            DDQ tmp = lMap.get(key);
            tmp.prev.next = tmp.next;
            if(tmp.next!=null) tmp.next.prev = tmp.prev;
            lMap.remove(key);
            length--;
        }

        DDQ newNode = new DDQ(key,value);
        lMap.put(key,newNode);
        
        DDQ tmp = dTail.prev;
        tmp.next = newNode;
        newNode.prev = tmp;
        newNode.next = dTail;
        dTail.prev = newNode;

        length++;
    }
}
```
