就是翻转链表，每遍历k个链表后记录一下起始点和终止点，然后把这一段链表扔到翻转链表代码里翻转一下就可以

```java
class Solution {
    public ListNode reverseKGroup(ListNode head, int k) {
        ListNode start=null;
        ListNode end=null;
        ListNode dummyHead = new ListNode();
        ListNode cur = head;
        int count=0;

        dummyHead.next = head;
        start = dummyHead;

        while(cur!=null){
            count++;
            if(count==k){
                end = cur;
                ListNode tmp = start.next;
                reverse(start,end, end.next);
                start = tmp;
                cur = start.next;
                count=0;
            }else{
                cur = cur.next;
            }
        }

        return dummyHead.next;
    }

    public void reverse(ListNode start,ListNode end,ListNode finalEnd){
         ListNode pre = end.next;
         ListNode cur = start.next;
         ListNode tmp;
         start.next = end;

         while(cur!=finalEnd){
            tmp = cur.next;
            cur.next = pre;
            pre = cur;
            cur = tmp;
        }
    }
}
```
