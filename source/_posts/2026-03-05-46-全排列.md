标准回溯，进入下一层回溯前选一个数字，然后把这个数字标记为以使用避免下一层内的过程中再次使用该数字，退出本层回溯后删除使用标记

```java
class Solution {
    Deque<Integer> path;
    List<List<Integer>> result;
    Map<Integer,Boolean> used;
    public List<List<Integer>> permute(int[] nums) {
        path = new LinkedList<>();
        result = new ArrayList<>();
        used = new LinkedHashMap<>();

        for(int i=0;i<=nums.length-1;i++){
            used.put(nums[i],false);
        }
        backtrace(nums);

        return result;
    }

    public void backtrace(int[] nums){
        if(path.size()==nums.length){
            result.add(new ArrayList<>(path));
            return;
        }
        for(int i=0;i<=nums.length-1;i++){
            boolean isUsed = used.get(nums[i]);
            if(!isUsed){
                path.add(nums[i]);
                used.put(nums[i],true);
                backtrace(nums);
                used.put(nums[i],false);
                path.removeLast();
            }
        }
    }
}
```
