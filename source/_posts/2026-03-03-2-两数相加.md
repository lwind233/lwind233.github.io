先区分长短链表

然后直接原地修改长链表的值，将和存储在长链表中，同时记录进位。

最后如果进位不为0则创建一个新的结点存储进位。

```java
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        int len1=0;
        int len2=0;
        int lenLong;
        int lenShort;
        ListNode cur = l1;
        ListNode shortL;
        ListNode longL;

        while(cur!=null){
            len1++;
            cur = cur.next;
        }

        cur = l2;
        while(cur!=null){
            len2++;
            cur = cur.next;
        }

        if(len1>=len2){
            longL = l1;
            shortL = l2;
            lenLong=len1;
            lenShort=len2;           
        }else{
            longL = l2;
            shortL = l1;
            lenLong=len2;
            lenShort=len1;           
        }

        ListNode shortCur = shortL;
        ListNode longCur = longL;
        ListNode tmp=longCur;
        int over = 0;
        int sum=0;
        while(longCur!=null){
            if(shortCur!=null) sum = shortCur.val+longCur.val+over;
            else sum = longCur.val+over;
            longCur.val = sum % 10;
            over = sum / 10;
            tmp = longCur;
            if(shortCur!=null) shortCur = shortCur.next;
            longCur = longCur.next;
        }
        if(over!=0){
            ListNode tmpNode = new ListNode();
            tmpNode.val = over;
            tmp.next = tmpNode;
        }

        return longL;
    }
}
```
